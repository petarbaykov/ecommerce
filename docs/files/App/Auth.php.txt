namespace Ecommerce\App;

use Ecommerce\Models\User;
use Ecommerce\App\Session;

/**
 * Class for the base Auth
 */
class Auth {
    
    
    
    public static function user() {
        $user = null;
        if(Session::has('user')) {
            $user = new User();
            $user = $user->find(Session::get('user'));
        }
        return $user;
    }
}
